69  subroutine hop_mod_init
70 
71  Implicit none
72 
73  Integer :: nc, nf, i,j
74  Complex (Kind=Kind(0.d0)) :: g
75 
76  ncheck = size(op_t,1)
77  If ( size(op_t,2) /= n_fl ) then
78  Write(6,*) 'Error in the number of flavors.'
79  stop
80  Endif
81  ndim_hop = op_t(1,1)%N
82  !Write(6,*) 'In Hop_mod: ', Ndim, Ndim_hop, Ncheck
83  Do nc = 1, ncheck
84  do nf = 1,n_fl
85  if ( ndim_hop /= op_t(nc,nf)%N ) Then
86  Write(6,*) 'Different size of Hoppings not implemented '
87  stop
88  endif
89  enddo
90  enddo
91 
92  Allocate ( exp_t(ndim_hop,ndim_hop,ncheck,n_fl) )
93  Allocate ( exp_t_m1(ndim_hop,ndim_hop,ncheck,n_fl) )
94  Allocate ( exp_t_1d2(ndim_hop,ndim_hop,ncheck,n_fl) )
95  Allocate ( exp_t_m1_1d2(ndim_hop,ndim_hop,ncheck,n_fl) )
96 
97  Allocate ( v_hlp(ndim_hop,ndim) )
98  Allocate ( v_hlp1(ndim_hop,ndim) )
99  Allocate ( u_hlp(ndim, ndim_hop) )
100  Allocate ( u_hlp1(ndim, ndim_hop) )
101 
102  exp_t = cmplx(0.d0, 0.d0, kind(0.d0))
103  exp_t_m1 = cmplx(0.d0, 0.d0, kind(0.d0))
104  do nf = 1,n_fl
105  do nc = 1,ncheck
106  g = op_t(nc,nf)%g
107  Call op_exp(g,op_t(nc,nf),exp_t(:,:,nc,nf))
108  g = -op_t(nc,nf)%g
109  Call op_exp(g,op_t(nc,nf),exp_t_m1(:,:,nc,nf))
110  g = op_t(nc,nf)%g/2.d0
111  Call op_exp(g,op_t(nc,nf),exp_t_1d2(:,:,nc,nf))
112  g = -op_t(nc,nf)%g/2.d0
113  Call op_exp(g,op_t(nc,nf),exp_t_m1_1d2(:,:,nc,nf))
114  ! symmetrize the upper part of Exp_T and Exp_T_M1
115  DO i = 1, ndim_hop
116  DO j = i, ndim_hop
117  exp_t(i, j, nc, nf) = (exp_t(i, j, nc, nf) + conjg(exp_t(j, i, nc, nf)))/2.d0
118  exp_t_m1(i, j, nc, nf) = (exp_t_m1(i, j, nc, nf) + conjg(exp_t_m1(j, i, nc, nf)))/2.d0
119  ENDDO
120  ENDDO
121  enddo
122  enddo
123 
124  zero = 1.e-12
125 
126  end subroutine hop_mod_init
