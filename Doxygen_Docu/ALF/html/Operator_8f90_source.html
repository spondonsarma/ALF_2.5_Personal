52  Type operator
53  Integer :: n, n_non_zero
54  logical :: diag
55  complex (Kind=Kind(0.d0)), pointer :: o(:,:), u (:,:), m_exp(:,:,:), e_exp(:,:)
56  Real (Kind=Kind(0.d0)), pointer :: e(:)
57  Integer, pointer :: p(:)
58  complex (Kind=Kind(0.d0)) :: g
59  complex (Kind=Kind(0.d0)) :: alpha
60  Integer :: Type
61  ! P is an N X Ndim matrix such that P.T*O*P* = A
62  ! P has only one non-zero entry per column which is specified by P
63  ! All in all. g * Phi(s,type) * ( c^{dagger} A c + alpha )
64  ! The variable Type allows you to define the type of HS.
65  ! The first N_non_zero elemets of diagonal matrix E are non-zero. The rest vanish.
66 
67  ! !!!!! M_exp and E_exp are for storage !!!!!
68  ! If Type =1 then the Ising field takes the values s = +/- 1
69  ! and M_exp has dimensions M_exp(N,N,-1:1)
70  ! If Type =2 then the Ising field takes the values s = +/- 1, +/- 2
71  ! and M_exp has dimensions M_exp(N,N,-2:2)
72  ! M_exp(:,:,s) = e^{g * Phi(s,type) * O(:,:) }
73  !
74  !
75  ! E_exp(:,s) = e^{g * Phi(s,type) *E(:) } and has dimensions E_exp(N,-1:1) for Type = 1
76  ! and dimensions E_exp(N,-2:2) for Type = 2
77  !
78  ! !!! If Type .neq. 1,2 then E_exp and M_exp are not allocated !!!
79 
80  end type operator

229  subroutine op_set(Op)
230  Implicit none
231  type(operator), intent(INOUT) :: op
232 
233  Complex (Kind=Kind(0.d0)), allocatable :: U(:,:), TMP(:, :)
234  Real (Kind=Kind(0.d0)), allocatable :: E(:)
235  Real (Kind=Kind(0.d0)) :: Zero = 1.d-9 !, Phi(-2:2)
236  Integer :: N, I, J, np,nz
237  Complex (Kind=Kind(0.d0)) :: Z
238  class(fields), allocatable :: nsigma_single
239 
240 
241  Allocate (nsigma_single)
242 
243  Call nsigma_single%make(1,1)
244 
245 
246  n = op%N
247  Allocate ( op%U(n,n), op%E(n) )
248  op%U = cmplx(0.d0, 0.d0, kind(0.d0))
249  op%E = 0.d0
250 
251  If (op%N > 1) then
252  n = op%N
253  op%diag = .true.
254  do i=1,n
255  do j=i+1,n
256  ! Binary comparison is OK here as Op%O was initialized to zero during Op_make.
257  if (op%O(i,j) .ne. cmplx(0.d0,0.d0, kind(0.d0)) .or. op%O(j,i) .ne. cmplx(0.d0,0.d0, kind(0.d0))) op%diag=.false.
258  enddo
259  enddo
260  if (op%diag) then
261  do i=1,n
262  op%E(i)=dble(op%O(i,i))
263  op%U(i,i)=cmplx(1.d0,0.d0, kind(0.d0))
264  enddo
265  op%N_non_zero = n
266  ! FFA Why do we assume that Op%N_non_zero = N for a diagonal operator?
267  else
268  Allocate (u(n,n), e(n), tmp(n, n))
269  Call diag(op%O,u, e)
270  np = 0
271  nz = 0
272  do i = 1,n
273  if ( abs(e(i)) > zero ) then
274  np = np + 1
275  op%U(:, np) = u(:, i)
276  op%E(np) = e(i)
277  else
278  op%U(:, n-nz) = u(:, i)
279  op%E(n-nz) = e(i)
280  nz = nz + 1
281  endif
282  enddo
283  op%N_non_zero = np
284  ! Write(6,*) "Op_set", np,N
285  tmp = op%U ! that way we have the changes to the determinant due to the permutation
286  z = det_c(tmp, n)
287  ! Scale Op%U to be in SU(N)
288  DO i = 1, n
289  op%U(i,1) = op%U(i, 1)/z
290  ENDDO
291  deallocate (u, e, tmp)
292  ! Op%U,Op%E)
293  ! Write(6,*) 'Calling diag 1'
294  endif
295  else
296  op%E(1) = REAL(Op%O(1,1), kind(0.D0))
297  op%U(1,1) = cmplx(1.d0, 0.d0 , kind(0.d0))
298  op%N_non_zero = 1
299  op%diag = .true.
300  endif
301  select case(op%type)
302  case(1)
303  Allocate(op%E_exp(op%N, -op%type : op%type), op%M_exp(op%N, op%N, -op%type : op%type))
304  nsigma_single%t(1) = 1
305  Do i=1,op%type
306  nsigma_single%f(1,1) = real(I,kind=kind(0.d0))
307  do n = 1, op%N
308  op%E_exp(n,i) = cmplx(1.d0, 0.d0, kind(0.d0))
309  op%E_exp(n,-i) = cmplx(1.d0, 0.d0, kind(0.d0))
310  if ( n <= op%N_non_Zero) then
311  !Op%E_exp(n,I) = exp(Op%g*Op%E(n)*Phi_st(I,1))
312  op%E_exp(n,i) = exp(op%g*op%E(n)*nsigma_single%Phi(1,1))
313  op%E_exp(n,-i) = 1.d0/op%E_exp(n,i)
314  endif
315  enddo
316  !call Op_exp(Op%g*Phi_st( I,1),Op,Op%M_exp(:,:,I))
317  !call Op_exp(Op%g*Phi_st(-I,1),Op,Op%M_exp(:,:,-I))
318  call op_exp( op%g*nsigma_single%Phi(1,1),op,op%M_exp(:,:,i))
319  call op_exp(-op%g*nsigma_single%Phi(1,1),op,op%M_exp(:,:,-i))
320  enddo
321  case(2)
322  Allocate(op%E_exp(op%N, -op%type : op%type), op%M_exp(op%N, op%N, -op%type : op%type))
323  nsigma_single%t(1) = 2
324  Do i=1,op%type
325  nsigma_single%f(1,1) = real(I,kind=kind(0.d0))
326  do n = 1, op%N
327  op%E_exp(n,i) = cmplx(1.d0, 0.d0, kind(0.d0))
328  op%E_exp(n,-i) = cmplx(1.d0, 0.d0, kind(0.d0))
329  if ( n <= op%N_non_Zero) then
330  !Op%E_exp(n,I) = exp(Op%g*Op%E(n)*Phi_st(I,2))
331  op%E_exp(n,i) = exp(op%g*op%E(n)*nsigma_single%Phi(1,1))
332  op%E_exp(n,-i) = 1.d0/op%E_exp(n,i)
333  endif
334  enddo
335  call op_exp( op%g*nsigma_single%Phi(1,1),op,op%M_exp(:,:,i))
336  call op_exp(-op%g*nsigma_single%Phi(1,1),op,op%M_exp(:,:,-i))
337  !call Op_exp(Op%g*Phi_st( I,2),Op,Op%M_exp(:,:,I))
338  !call Op_exp(Op%g*Phi_st(-I,2),Op,Op%M_exp(:,:,-I))
339  enddo
340  case default
341  end select
342 
343  Call nsigma_single%clear()
344 
345  Deallocate (nsigma_single)
346 
347  end subroutine op_set

367 
368  Implicit none
369 
370  type(operator), Intent(IN) :: op
371  Complex (Kind=Kind(0.d0)), Dimension(:,:), INTENT(OUT) :: Mat
372  Complex (Kind=Kind(0.d0)), INTENT(IN) :: g
373  Complex (Kind=Kind(0.d0)) :: Z, Z1, y, t
374  Complex (Kind=Kind(0.d0)), allocatable, dimension(:,:) :: c
375 
376  Integer :: n, j, I, iters
377 
378  iters = op%N
379  mat = cmplx(0.d0, 0.d0, kind(0.d0))
380  if (op%diag) then
381  Do n = 1, iters
382  mat(n,n)=exp(g*op%E(n))
383  enddo
384  else
385  Allocate (c(iters, iters))
386  c = 0.d0
387  Do n = 1, iters
388  z = exp(g*op%E(n))
389  do j = 1, iters
390  z1 = z*conjg(op%U(j,n))
391  do i = 1, iters
392  ! This performs Kahan summation so as to improve precision.
393  y = z1 * op%U(i, n) - c(i, j)
394  t = mat(i, j) + y
395  c(i, j) = (t - mat(i,j)) - y
396  mat(i, j) = t
397  ! Mat(I, J) = Mat(I, J) + Z1 * Op%U(I, n)
398  enddo
399  ! Mat(1:iters, J) = Mat(1:iters, J) + Z1 * Op%U(1:iters, n)
400  enddo
401  enddo
402  Deallocate(c)
403  endif
404  end subroutine op_exp
