111  Integer , INTENT(IN) :: N_op, Nt, Op_dim
112  Complex (Kind=Kind(0.d0)), INTENT(INOUT) :: Phase
113  Integer , INTENT(IN) :: ns_new

123  Complex (Kind=Kind(0.d0)) :: Ratio(N_FL), Ratiotot, Z1
124  Integer :: n,m,nf, i
125  Complex (Kind=Kind(0.d0)) :: Z, D_Mat, myexp, s1, s2

128  Complex (Kind=Kind(0.d0)) :: u(Ndim,Op_dim), v(Ndim,Op_dim) ,alpha, beta

134  toggle = .false.
135  ! if ( abs(OP_V(n_op,1)%g) < 1.D-12 ) return

137  Allocate (nsigma_new)
138  Call nsigma_new%make(1,1)

140  ! Compute the ratio
141  nf = 1
142  nsigma_new%f(1,1) = real(ns_new,kind=kind(0.d0))
143  nsigma_new%t(1) = op_v(n_op,nf)%Type
144  Do nf = 1,n_fl
145  !Z1 = Op_V(n_op,nf)%g * ( Phi_st(ns_new,Op_V(n_op,nf)%type) - Phi_st(ns_old,Op_V(n_op,nf)%type))
146  z1 = op_v(n_op,nf)%g * ( nsigma_new%Phi(1,1) - nsigma%Phi(n_op,nt) )
147  Do m = 1,op_v(n_op,nf)%N_non_zero
148  myexp = exp( z1* op_v(n_op,nf)%E(m) )
149  z = myexp - 1.d0
150  delta(m,nf) = z
151  do n = 1,op_v(n_op,nf)%N_non_zero
152  mat(n,m) = - z * gr( op_v(n_op,nf)%P(n), op_v(n_op,nf)%P(m),nf )
153  Enddo
154  mat(m,m) = myexp + mat(m,m)
155  Enddo
156  If (Size(mat,1) == 1 ) then
157  d_mat = mat(1,1)
158  elseif (Size(mat,1) == 2 ) then
159  s1 = mat(1,1)*mat(2,2)
160  s2 = mat(2,1)*mat(1,2)
161  If (abs(s1) > abs(s2)) then
162  d_mat = s1*(1.d0 - s2/s1)
163  else
164  d_mat = s2*(s1/s2 - 1.d0)
165  Endif
166  ! D_mat = - Mat(2,1)*Mat(1,2)
167  else
168  d_mat = det(mat,Size(mat,1))
169  endif
170  ratio(nf) = d_mat * exp( z1*op_v(n_op,nf)%alpha )
171  Enddo
172 
173  ratiotot = product(ratio)
174  nf = 1
175  !Ratiotot = (Ratiotot**dble(N_SUN)) * Gama_st(ns_new, Op_V(n_op,nf)%type)/Gama_st(ns_old, Op_V(n_op,nf)%type)
176  ratiotot = (ratiotot**dble(n_sun)) * nsigma_new%Gama(1,1)/nsigma%Gama(n_op,nt)
177 
178  !Write(6,*) Ratiotot
179 
180  If (mode == "Final" ) Then
181  !Write(6,*) "Up_I: ", Ratiotot
182  ratiotot = ratiotot * prev_ratiotot
183  !Write(6,*) "Up_f: ", Ratiotot
184  weight = s0_ratio * t0_proposal_ratio * abs( real(Phase * Ratiotot, kind=kind(0.d0))/real(Phase,kind=Kind(0.d0)) )
185  !Write(6,*) Phase, Prev_Ratiotot, S0_ratio, T0_proposal_ratio, ns_old,ns_new
186  elseif (mode == "Intermediate" ) Then
187  weight = 1.5d0
188  !Write(6,*) "Up_I: ", Ratiotot
189  prev_ratiotot = prev_ratiotot*ratiotot
190  else
191  Write(6,*) 'Error'
192  stop
193  endif
194 
195  toggle = .false.
196  if ( weight > ranf_wrap() ) Then
197  toggle = .true.
198  If (mode == "Final" ) phase = phase * ratiotot/sqrt(ratiotot*conjg(ratiotot))
199  !Write(6,*) 'Accepted : ', Ratiotot
200 
201  Do nf = 1,n_fl
202  ! Setup u(i,n), v(n,i)
203  beta = 0.d0
204  call zlaset('N', ndim, op_dim, beta, beta, u, size(u, 1))
205  call zlaset('N', ndim, op_dim, beta, beta, v, size(v, 1))
206  do n = 1,op_v(n_op,nf)%N_non_zero
207  u( op_v(n_op,nf)%P(n), n) = delta(n,nf)
208  do i = 1,ndim
209  v(i,n) = - gr( op_v(n_op,nf)%P(n), i, nf )
210  enddo
211  v(op_v(n_op,nf)%P(n), n) = 1.d0 - gr( op_v(n_op,nf)%P(n), op_v(n_op,nf)%P(n), nf)
212  enddo
213 
214  call zlaset('N', ndim, op_dim, beta, beta, x_v, size(x_v, 1))
215  call zlaset('N', ndim, op_dim, beta, beta, y_v, size(y_v, 1))
216  i = op_v(n_op,nf)%P(1)
217  x_v(i, 1) = u(i, 1)/(1.d0 + v(i,1)*u(i,1) )
218  call zcopy(ndim, v(:, 1), 1, y_v(:, 1), 1)
219  do n = 2,op_v(n_op,nf)%N_non_zero
220  call zcopy(ndim, u(:, n), 1, x_v(:, n), 1)
221  call zcopy(ndim, v(:, n), 1, y_v(:, n), 1)
222  z = 1.d0 + u( op_v(n_op,nf)%P(n), n)*v(op_v(n_op,nf)%P(n),n)
223  alpha = -1.d0
224  Allocate(syu(n), sxv(n))
225  call zgemv('T', ndim, n-1, alpha, y_v, ndim, u(1,n), 1, beta , syu, 1)
226  call zgemv('T', ndim, n-1, alpha, x_v, ndim, v(1,n), 1, beta , sxv, 1)
227  alpha = 1.d0
228  call zgemv('N', ndim, n-1, alpha, x_v, ndim, syu, 1, alpha, x_v(1, n), 1)
229  call zgemv('N', ndim, n-1, alpha, y_v, ndim, sxv, 1, alpha, y_v(1, n), 1)
230  do m = 1,n-1
231  z = z - syu(m)*sxv(m)
232  enddo
233  z = 1.d0/z
234  call zscal(ndim, z, x_v(1, n), 1)
235  Deallocate(syu, sxv)
236  enddo
237  IF (size(op_v(n_op,nf)%P, 1) == 1) THEN
238  CALL zcopy(ndim, gr(1, op_v(n_op,nf)%P(1), nf), 1, xp_v(1, 1), 1)
239  z = -x_v(op_v(n_op,nf)%P(1), 1)
240  CALL zgeru(ndim, ndim, z, xp_v(1,1), 1, y_v(1, 1), 1, gr(1,1,nf), ndim)
241  ELSE
242  Allocate (zarr(size(op_v(n_op,nf)%P, 1), op_dim), grarr(ndim, op_dim))
243  zarr = x_v(op_v(n_op,nf)%P, :)
244  grarr = gr(:, op_v(n_op,nf)%P, nf)
245  alpha = 1.d0
246  CALL zgemm('N', 'N', ndim, op_dim, op_dim, alpha, grarr, ndim, zarr, size(op_v(n_op,nf)%P, 1), beta, xp_v, ndim)
247  Deallocate(zarr, grarr)
248  beta = cmplx( 1.0d0, 0.0d0, kind(0.d0))
249  alpha = -1.d0
250  CALL zgemm('N','T',ndim,ndim,op_dim,alpha,xp_v, ndim, y_v, ndim,beta,gr(1,1,nf), ndim)
251  ENDIF
252 
253  enddo
254 
255  ! Flip the spin
256  nsigma%f(n_op,nt) = nsigma_new%f(1,1) ! real(ns_new,Kind=kind(0.d0))
257  endif
258 
259  If ( mode == "Final" ) then
260  Call control_upgrade(toggle)
261  Call control_upgrade_eff(toggle)
262  endif
263 
264  Call nsigma_new%clear()
265  Deallocate (nsigma_new)
