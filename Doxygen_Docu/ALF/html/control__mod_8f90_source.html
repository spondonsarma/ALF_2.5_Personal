308  subroutine make_truncation(prog_truncation,cpu_max,count_bin_start,count_bin_end)
309  !!!!!!! Written by M. Bercx
310  ! This subroutine checks if the conditions for a controlled termination of the program are met.
311  ! The subroutine contains a hard-coded threshold (in unit of bins):
312  ! if time_remain/time_bin_duration < threshold the program terminates.
313 
314 #ifdef MPI
315  Use mpi
316 #endif
317 
318  logical, intent(out) :: prog_truncation
319  real(kind=kind(0.d0)), intent(in) :: cpu_max
320  integer(kind=kind(0.d0)), intent(in) :: count_bin_start, count_bin_end
321  real(kind=kind(0.d0)) :: count_alloc_end
322  real(kind=kind(0.d0)) :: time_bin_duration,time_remain,bins_remain,threshold
323 #ifdef MPI
324  real(kind=kind(0.d0)) :: bins_remain_mpi
325  integer :: err_mpi,rank_mpi,tasks_mpi
326 #endif
327  threshold = 1.5d0
328  prog_truncation = .false.
329 
330 #ifdef MPI
331  call mpi_comm_size(mpi_comm_world, tasks_mpi, err_mpi)
332  call mpi_comm_rank(mpi_comm_world, rank_mpi, err_mpi)
333 #endif
334  count_alloc_end = count_cpu_start + cpu_max*3600*count_rate
335  time_bin_duration = (count_bin_end-count_bin_start)/dble(count_rate)
336  time_remain = (count_alloc_end - count_bin_end)/dble(count_rate)
337  if (count_bin_end .lt. count_bin_start) then ! the counter has wrapped around
338  time_bin_duration = (count_max+count_bin_end-count_bin_start)/dble(count_rate)
339  time_remain = (count_alloc_end - count_bin_end-count_max)/dble(count_rate)
340  endif
341  bins_remain = time_remain/time_bin_duration
342 
343 #ifdef MPI
344  call mpi_reduce(bins_remain,bins_remain_mpi,1,mpi_double_precision,mpi_sum,0,mpi_comm_world,err_mpi)
345 #endif
346 
347 #ifdef MPI
348  if (rank_mpi .eq. 0) bins_remain_mpi = bins_remain_mpi/tasks_mpi
349  call mpi_bcast(bins_remain_mpi,1, mpi_double_precision,0, mpi_comm_world,err_mpi)
350  if (bins_remain_mpi .lt. threshold) prog_truncation = .true.
351 #else
352  if (bins_remain .lt. threshold) prog_truncation = .true.
353 #endif
354  end subroutine make_truncation
