296 SUBROUTINE assign_udv_state(this, src)
297  IMPLICIT NONE
298  CLASS(UDV_State), INTENT(INOUT) :: this
299  CLASS(UDV_State), INTENT(IN) :: src
300 
301  IF(this%ndim .ne. src%ndim .or. this%n_part .ne. src%n_part) call this%dealloc
302  this%ndim = src%ndim
303  this%n_part = src%n_part
304  this%side = src%side
305 
306  IF(.not. ALLOCATED(this%U)) ALLOCATE(this%U(this%ndim, this%n_part))
307  IF(.not. ALLOCATED(this%V) .and. ALLOCATED(src%V)) ALLOCATE(this%V(this%n_part, this%n_part))
308  associate(ndim => src%ndim)
309  CALL zlacpy('A', ndim, this%n_part, src%U(1, 1), ndim, this%U(1, 1), ndim)
310  if (ALLOCATED(src%V)) CALL zlacpy('A', this%n_part, this%n_part, src%V(1, 1), &
311  & this%n_part, this%V(1, 1), this%n_part)
312  END associate
313 #if !defined(LOG)
314  IF(.not. ALLOCATED(this%D)) ALLOCATE(this%D(this%n_part))
315  this%D = src%D
316 #else
317  IF(.not. ALLOCATED(this%L)) ALLOCATE(this%L(this%n_part))
318  this%L = src%L
319 #endif
320 END SUBROUTINE assign_udv_state

447  SUBROUTINE decompose_udv_state(UDVR)!, TMP, TMP1, NCON)
448  Use qdrp_mod
449  Use mymats
450  Implicit None
451 ! INTEGER, intent(in) :: NCON
452 ! COMPLEX (Kind=Kind(0.d0)), intent(in), allocatable, dimension(:, :) :: TMP
453 ! COMPLEX (Kind=Kind(0.d0)), intent(inout), allocatable, dimension(:, :) :: TMP1
454  CLASS(UDV_State), intent(inout) :: UDVR
455  COMPLEX (Kind=Kind(0.d0)), allocatable, Dimension(:) :: TAU, WORK
456  COMPLEX (Kind=Kind(0.d0)) :: Z_ONE, beta, phase
457  INTEGER :: INFO, i, LWORK, Ndim, N_part
458  INTEGER, allocatable, Dimension(:) :: IPVT
459 #ifdef LOG
460  REAL (Kind=Kind(0.d0)), allocatable, Dimension(:) :: tmpnorm
461  REAL (Kind=Kind(0.d0)) :: tmpL, DZNRM2
462  INTEGER :: J, PVT
463  COMPLEX (Kind=Kind(0.d0)), allocatable, Dimension(:) :: D
464 #else
465  LOGICAL :: FORWRD
466 #endif
467 
468 ! if(udvr%side .ne. "R" .and. udvr%side .ne. "r" ) then
469 ! write(*,*) "calling wrong decompose"
470 ! endif
471 
472  ! QR(TMP * U * D) * V
473  z_one = cmplx(1.d0, 0.d0, kind(0.d0))
474  ndim = udvr%ndim
475  n_part = udvr%n_part
476  ALLOCATE(tau(n_part), ipvt(n_part))
477 #if !defined(LOG)
478  ! TMP1 = TMP1 * D
479  If( ALLOCATED(udvr%V) ) then
480  DO i = 1,n_part
481  udvr%U(:, i) = udvr%U(:, i)*udvr%D(i)
482  ENDDO
483  endif
484  !use lapack internal pivoting
485  ipvt = 0
486  call qdrp_decompose(ndim, n_part, udvr%U, udvr%D, ipvt, tau, work, lwork)
487  phase=cmplx(1.d0,0.d0,kind(0.d0))
488  do i=1,n_part
489  phase=phase*udvr%U(i,i)
490  enddo
491  Call pivot_phase(phase,ipvt,n_part)
492  if(udvr%side == "L" .or. udvr%side == "l" ) then
493  phase=conjg(phase)
494  endif
495  beta=1/phase
496  If( ALLOCATED(udvr%V) ) then
497  !scale first row of R with 1/phase to set Det(R)=1 [=Det(V)]
498  call zscal(n_part,beta,udvr%U(1,1),ndim)
499  ! Permute V. Since we multiply with V from the right we have to permute the rows.
500  ! A V = A P P^-1 V = Q R P^-1 V
501  forwrd = .true.
502  if(udvr%side == "R" .or. udvr%side == "r" ) then
503  CALL zlapmr(forwrd, n_part, n_part, udvr%V, n_part, ipvt(1)) ! lapack 3.3
504  else
505  CALL zlapmt(forwrd, n_part, n_part, udvr%V, n_part, ipvt(1))
506  endif
507  endif
508 #else
509  !manually perform pivoting (using the logscale if LOG is defined)
510  ALLOCATE(tmpnorm(n_part),d(n_part))
511  if ( ALLOCATED(udvr%V) ) then
512  Do i=1,n_part
513  tmpnorm(i) = log(dznrm2( ndim, udvr%U( 1, i ), 1 ))+udvr%L(i)
514  enddo
515  else
516  Do i=1,n_part
517  tmpnorm(i) = log(dznrm2( ndim, udvr%U( 1, i ), 1 ))
518  enddo
519  endif
520 ! TmpMat=UDVr%V
521 ! phase=det_c(tmpmat,ndim)
522 ! write(*,*) "Phase in:",phase
523  phase=cmplx(1.d0,0.d0,kind(0.d0))
524  do i=1,n_part
525  pvt = i
526  do j=i+1,n_part
527  if( tmpnorm(j)>tmpnorm(pvt) ) pvt=j
528  enddo
529  ipvt(i)=pvt
530  IF( pvt.NE.i ) THEN
531  CALL zswap( ndim, udvr%U( 1, pvt ), 1, udvr%U( 1, i ), 1 )
532  If( ALLOCATED(udvr%V) ) then
533  if(udvr%side == "R" .or. udvr%side == "r" ) then
534  CALL zswap( n_part, udvr%V( pvt, 1 ), n_part, udvr%V( i, 1 ), n_part )
535  else
536  CALL zswap( n_part, udvr%V( 1, pvt ), 1, udvr%V( 1, i ), 1 )
537  endif
538  endif
539  tmpl=udvr%L(i)
540  udvr%L(i)=udvr%L(pvt)
541  udvr%L(pvt)=tmpl
542  tmpnorm( pvt ) = tmpnorm( i )
543  phase=-phase
544  END IF
545  enddo
546  !disable lapack internal pivoting
547  ipvt = 1
548  call qdrp_decompose(ndim, n_part, udvr%U, d, ipvt, tau, work, lwork)
549  do i=1,n_part
550  phase=phase*udvr%U(i,i)
551  enddo
552  if(udvr%side == "L" .or. udvr%side == "l" ) then
553  phase=conjg(phase)
554  endif
555  beta=1/phase
556  If( ALLOCATED(udvr%V) ) then
557  !scale first row of R with 1/phase to set Det(R)=1 [=Det(V)]
558  call zscal(n_part,beta,udvr%U(1,1),ndim)
559  do i=1,n_part
560  do j=i+1,n_part
561  udvr%U(i,j)=udvr%U(i,j)*cmplx(exp(udvr%L(j)-udvr%L(i)),0.d0,kind(0.d0))
562  enddo
563  udvr%L(i)=log(dble(d(i))) + udvr%L(i)
564  enddo
565  else
566  do i=1,n_part
567  udvr%L(i)=log(dble(d(i)))
568  enddo
569  endif
570  DEALLOCATE(d, tmpnorm)
571 #endif
572  If( ALLOCATED(udvr%V) ) then
573  if(udvr%side == "R" .or. udvr%side == "r" ) then
574  ! V = R * V
575  CALL ztrmm('L', 'U', 'N', 'N', n_part, n_part, z_one, udvr%U, ndim, udvr%V, n_part)
576  else
577  ! V = V * R^dagger
578  CALL ztrmm('R', 'U', 'C', 'N', n_part, n_part, z_one, udvr%U, ndim, udvr%V, n_part)
579  endif
580  endif
581  ! Generate explicitly U in the previously abused storage of U
582  CALL zungqr(ndim, n_part, n_part, udvr%U, ndim, tau, work, lwork, info)
583  ! scale first column of U to correct the scaling in V such that UDV is not changed
584  call zscal(ndim,phase,udvr%U(1,1),1)
585  DEALLOCATE(tau, work, ipvt)
586 END SUBROUTINE decompose_udv_state
