394  SUBROUTINE decompose_udv_state(UDVR)
395  Use qdrp_mod
396  Use mymats
397  Implicit None
398  CLASS(UDV_State), intent(inout) :: UDVR
399  COMPLEX (Kind=Kind(0.d0)), allocatable, Dimension(:) :: TAU, WORK
400  COMPLEX (Kind=Kind(0.d0)) :: Z_ONE, beta, phase
401  INTEGER :: INFO, i, LWORK, Ndim, N_part
402  INTEGER, allocatable, Dimension(:) :: IPVT
403 #ifdef LOG
404  REAL (Kind=Kind(0.d0)), allocatable, Dimension(:) :: tmpnorm
405  REAL (Kind=Kind(0.d0)) :: tmpL, DZNRM2
406  INTEGER :: J, PVT
407  COMPLEX (Kind=Kind(0.d0)), allocatable, Dimension(:) :: D
408 #else
409  LOGICAL :: FORWRD
410 #endif
411 
412  ! QR(TMP * U * D) * V
413  z_one = cmplx(1.d0, 0.d0, kind(0.d0))
414  ndim = udvr%ndim
415  n_part = udvr%n_part
416  ALLOCATE(tau(n_part), ipvt(n_part))
417 #if !defined(LOG)
418  ! TMP1 = TMP1 * D
419  If( ALLOCATED(udvr%V) ) then
420  DO i = 1,n_part
421  udvr%U(:, i) = udvr%U(:, i)*udvr%D(i)
422  ENDDO
423  endif
424  !use lapack internal pivoting
425  ipvt = 0
426  call qdrp_decompose(ndim, n_part, udvr%U, udvr%D, ipvt, tau, work, lwork)
427  phase=cmplx(1.d0,0.d0,kind(0.d0))
428  do i=1,n_part
429  phase=phase*udvr%U(i,i)
430  enddo
431  Call pivot_phase(phase,ipvt,n_part)
432  if(udvr%side == "L" .or. udvr%side == "l" ) then
433  phase=conjg(phase)
434  endif
435  beta=1/phase
436  If( ALLOCATED(udvr%V) ) then
437  !scale first row of R with 1/phase to set Det(R)=1 [=Det(V)]
438  call zscal(n_part,beta,udvr%U(1,1),ndim)
439  ! Permute V. Since we multiply with V from the right we have to permute the rows.
440  ! A V = A P P^-1 V = Q R P^-1 V
441  forwrd = .true.
442  if(udvr%side == "R" .or. udvr%side == "r" ) then
443  CALL zlapmr(forwrd, n_part, n_part, udvr%V, n_part, ipvt(1)) ! lapack 3.3
444  else
445  CALL zlapmt(forwrd, n_part, n_part, udvr%V, n_part, ipvt(1))
446  endif
447  endif
448 #else
449  !manually perform pivoting (using the logscale if LOG is defined)
450  ALLOCATE(tmpnorm(n_part),d(n_part))
451  if ( ALLOCATED(udvr%V) ) then
452  Do i=1,n_part
453  tmpnorm(i) = log(dznrm2( ndim, udvr%U( 1, i ), 1 ))+udvr%L(i)
454  enddo
455  else
456  Do i=1,n_part
457  tmpnorm(i) = log(dznrm2( ndim, udvr%U( 1, i ), 1 ))
458  enddo
459  endif
460  ! TmpMat=UDVr%V
461  ! phase=det_c(tmpmat,ndim)
462  ! write(*,*) "Phase in:",phase
463  phase=cmplx(1.d0,0.d0,kind(0.d0))
464  do i=1,n_part
465  pvt = i
466  do j=i+1,n_part
467  if( tmpnorm(j)>tmpnorm(pvt) ) pvt=j
468  enddo
469  ipvt(i)=pvt
470  IF( pvt.NE.i ) THEN
471  CALL zswap( ndim, udvr%U( 1, pvt ), 1, udvr%U( 1, i ), 1 )
472  If( ALLOCATED(udvr%V) ) then
473  if(udvr%side == "R" .or. udvr%side == "r" ) then
474  CALL zswap( n_part, udvr%V( pvt, 1 ), n_part, udvr%V( i, 1 ), n_part )
475  else
476  CALL zswap( n_part, udvr%V( 1, pvt ), 1, udvr%V( 1, i ), 1 )
477  endif
478  endif
479  tmpl=udvr%L(i)
480  udvr%L(i)=udvr%L(pvt)
481  udvr%L(pvt)=tmpl
482  tmpnorm( pvt ) = tmpnorm( i )
483  phase=-phase
484  END IF
485  enddo
486  !disable lapack internal pivoting
487  ipvt = 1
488  call qdrp_decompose(ndim, n_part, udvr%U, d, ipvt, tau, work, lwork)
489  do i=1,n_part
490  phase=phase*udvr%U(i,i)
491  enddo
492  if(udvr%side == "L" .or. udvr%side == "l" ) then
493  phase=conjg(phase)
494  endif
495  beta=1/phase
496  If( ALLOCATED(udvr%V) ) then
497  !scale first row of R with 1/phase to set Det(R)=1 [=Det(V)]
498  call zscal(n_part,beta,udvr%U(1,1),ndim)
499  do i=1,n_part
500  do j=i+1,n_part
501  udvr%U(i,j)=udvr%U(i,j)*cmplx(exp(udvr%L(j)-udvr%L(i)),0.d0,kind(0.d0))
502  enddo
503  udvr%L(i)=log(dble(d(i))) + udvr%L(i)
504  enddo
505  else
506  do i=1,n_part
507  udvr%L(i)=log(dble(d(i)))
508  enddo
509  endif
510  DEALLOCATE(d, tmpnorm)
511 #endif
512  If( ALLOCATED(udvr%V) ) then
513  if(udvr%side == "R" .or. udvr%side == "r" ) then
514  ! V = R * V
515  CALL ztrmm('L', 'U', 'N', 'N', n_part, n_part, z_one, udvr%U, ndim, udvr%V, n_part)
516  else
517  ! V = V * R^dagger
518  CALL ztrmm('R', 'U', 'C', 'N', n_part, n_part, z_one, udvr%U, ndim, udvr%V, n_part)
519  endif
520  endif
521  ! Generate explicitly U in the previously abused storage of U
522  CALL zungqr(ndim, n_part, n_part, udvr%U, ndim, tau, work, lwork, info)
523  ! scale first column of U to correct the scaling in V such that UDV is not changed
524  call zscal(ndim,phase,udvr%U(1,1),1)
525 
526  DEALLOCATE(tau, work, ipvt)
527 
528  END SUBROUTINE decompose_udv_state
