60  SUBROUTINE qdrp_decompose(Ndim, N_part, Mat, D, IPVT, TAU, WORK, LWORK)
61  Implicit None
62  Integer, intent(in) :: Ndim
63  Integer, intent(in) :: N_part
64  Integer, intent(inout) :: LWORK
65  Integer, Dimension(:), intent(inout), Allocatable :: IPVT
66  COMPLEX(Kind=Kind(0.d0)), Dimension(:,:), Intent(inout) :: Mat
67  COMPLEX(Kind=Kind(0.d0)), Dimension(:), Intent(inout) :: D
68  COMPLEX(Kind=Kind(0.d0)), Dimension(:), Intent(inout), Allocatable :: TAU
69  COMPLEX(Kind=Kind(0.d0)), Dimension(:), Intent(INOUT), Allocatable :: WORK
70 
71  COMPLEX(Kind=Kind(0.d0)), Dimension(:), Allocatable :: RWORK
72  COMPLEX(Kind=Kind(0.d0)) :: Z
73  Integer :: info, i, j
74  Real(Kind=Kind(0.d0)) :: X
75 
76  ALLOCATE(rwork(2*ndim))
77  ! Query optimal amount of memory
78  call zgeqp3(ndim, n_part, mat(1,1), ndim, ipvt, tau(1), z, -1, rwork(1), info)
79  lwork = int(dble(z))
80  ALLOCATE(work(lwork))
81  ! QR decomposition of Mat with full column pivoting, Mat * P = Q * R
82  call zgeqp3(ndim, n_part, mat(1,1), ndim, ipvt, tau(1), work(1), lwork, rwork(1), info)
83  DEALLOCATE(rwork)
84  ! separate off D
85  do i = 1, n_part
86  ! plain diagonal entry
87  x = abs(mat(i, i))
88  ! ! a inf-norm
89  ! X = TPUP(i, i+izamax(Ndim+1-i, TPUP(i, i), Ndim)-1)
90  ! ! another inf-norm
91  ! X = TPUP(i, i-1+izmax1(Ndim+1-i, TPUP(i, i), Ndim))
92  ! ! 1-norm
93  ! X = DZSUM1(N_size+1-i, TPUP(i, i), N_size)
94  ! 2-norm
95  ! X = DZNRM2(N_size+1-i, TPUP(i, i), N_size)
96  d(i) = x
97  do j = i, n_part
98  mat(i, j) = mat(i, j) / x
99  enddo
100  enddo
101  END SUBROUTINE qdrp_decompose
