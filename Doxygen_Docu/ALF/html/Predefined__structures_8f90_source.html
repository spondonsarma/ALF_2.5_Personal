100  Subroutine predefined_latt(Lattice_type, L1,L2,Norb,N_coord,Ndim, List,Invlist,Latt)
101 
102  Implicit none
103 
104  !Set the lattice
105  Character (len=64), Intent(IN) :: Lattice_type

107  Integer, Intent(OUT) :: Norb,N_coord, Ndim
108  Integer, Intent(OUT), Dimension(:,:), allocatable :: List, Invlist
109  Type(Lattice), Intent(Out) :: Latt

111  Real (Kind=Kind(0.d0)) :: a1_p(2), a2_p(2), L1_p(2), L2_p(2)
112  Integer :: I, nc, no
113 
114  select case (lattice_type)
115  case("Square")
116  If (l1==1 .or. l2==1 ) then
117  Write(6,*) 'For one-dimensional lattices set : L2 = 1'
118  stop
119  endif
120  norb = 1
121  n_coord = 2
122  a1_p(1) = 1.0 ; a1_p(2) = 0.d0
123  a2_p(1) = 0.0 ; a2_p(2) = 1.d0
124  l1_p = dble(l1)*a1_p
125  l2_p = dble(l2)*a2_p
126  Call make_lattice( l1_p, l2_p, a1_p, a2_p, latt )
127  case("One_dimensional")
128  If (l1 == 1 ) then
129  Write(6,*) ' For one dimensional systems set L2 = 1 '
130  stop
131  endif
132  norb = 1
133  n_coord = 1
134  a1_p(1) = 1.0 ; a1_p(2) = 0.d0
135  a2_p(1) = 0.0 ; a2_p(2) = 1.d0
136  l1_p = dble(l1)*a1_p
137  l2_p = dble(l2)*a2_p
138  Call make_lattice( l1_p, l2_p, a1_p, a2_p, latt )
139  case("Honeycomb")
140  If (l1==1 .or. l2==1 ) then
141  Write(6,*) 'For one-dimensional lattices set : L2 = 1'
142  stop
143  endif
144  norb = 2
145  n_coord = 3
146  a1_p(1) = 1.d0 ; a1_p(2) = 0.d0
147  a2_p(1) = 0.5d0 ; a2_p(2) = sqrt(3.d0)/2.d0
148 
149  !del_p = (a2_p - 0.5*a1_p ) * 2.0/3.0
150  l1_p = dble(l1) * a1_p
151  l2_p = dble(l2) * a2_p
152  Call make_lattice( l1_p, l2_p, a1_p, a2_p, latt )
153  case("Pi_Flux")
154  If (l1==1 .or. l2==1 ) then
155  Write(6,*) 'For one-dimensional lattices set : L2 = 1'
156  stop
157  endif
158  norb = 2
159  n_coord = 4
160  a1_p(1) = 1.d0 ; a1_p(2) = 1.d0
161  a2_p(1) = 1.d0 ; a2_p(2) = -1.d0
162 
163  !del_p = (a2_p - 0.5*a1_p ) * 2.0/3.0
164  l1_p = dble(l1) * (a1_p - a2_p)/2.d0
165  l2_p = dble(l2) * (a1_p + a2_p)/2.d0
166  Call make_lattice( l1_p, l2_p, a1_p, a2_p, latt )
167  case default
168  Write(6,*) "Lattice not yet implemented!"
169  stop
170  end select
171  ! Call Print_latt(Latt)
172  ! This is for the orbital structure.
173 
174  ndim = latt%N*norb
175  Allocate (list(ndim,2), invlist(latt%N,norb))
176  nc = 0
177  Do i = 1,latt%N
178  Do no = 1,norb
179  ! For the Honeycomb and pi-flux lattices no = 1,2 corresponds to the A,and B sublattice.
180  nc = nc + 1
181  list(nc,1) = i
182  list(nc,2) = no
183  invlist(i,no) = nc
184  Enddo
185  Enddo
186 
187  end Subroutine predefined_latt
